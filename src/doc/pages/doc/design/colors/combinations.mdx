Utilizza la seguente guida per comprendere come e quando usare insieme i colori della nostra tavolozza.

<Usage>
  <UsageDo>
    <Caption>Usa qualsiasi colore di sfondo chiaro con il colore del testo corrispondente.</Caption>
    <Button className="background-color-brand-maggioli-14 color-brand-maggioli-01">Hello world</Button>
    <CodeSnippet language="scss" code={`.component {
  background-color: color('brand.maggioli', '14');
  color: color('brand.maggioli', '01');
}`} />
  </UsageDo>
  <UsageDont>
    <Caption>Controlla che i <Link href="/doc/design/colors/palette/">test di contrasto</Link> non falliscano.</Caption>
    <Button className="background-color-brand-maggioli-14 color-brand-maggioli-11">Hello world</Button>
    <CodeSnippet language="scss" code={`.component {
  background-color: color('brand.maggioli', '14');
  color: color('brand.maggioli', '11');
}`} />
  </UsageDont>
</Usage>
